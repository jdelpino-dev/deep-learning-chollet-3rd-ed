# Target Python: 3.11.14  (via pyenv + pyenv-virtualenv)
#
# Setup from scratch:
#   pyenv install 3.11.14
#   pyenv virtualenv 3.11.14 dl-chollet-3rd-tf
#   pyenv local dl-chollet-3rd-tf
#   pip install --upgrade pip
#   pip install -r requirements_tf.txt
#
# Strip notebook outputs from git (requires nbstripout via pipx):
#   pipx install nbstripout              # one-time, global install
#   nbstripout --install                  # per-repo git filter
#   git config filter.nbstripout.extrakeys \
#     'metadata.kernelspec metadata.language_info metadata.widgets cell.metadata.scrolled cell.metadata.collapsed'
#
# Recreating the env (if needed):
#   pyenv virtualenv-delete dl-chollet-3rd-tf -f
#   pyenv virtualenv 3.11.14 dl-chollet-3rd-tf
#   pyenv local dl-chollet-3rd-tf
#   pip install --upgrade pip
#   pip install -r requirements_tf.txt
#
# Verify:
#   python -c "import tensorflow as tf; print('TF', tf.__version__)"
#   python -c "import tensorflow as tf; print('GPU:', tf.config.list_physical_devices('GPU'))"
#   python -c "import keras; print('Keras', keras.__version__)"
#
# Why Python 3.11?
#   tensorflow-metal (Apple's GPU plug-in for macOS) only supports Python
#   up to 3.11.x. Python 3.12+ is incompatible with the current release
#   (tensorflow-metal 1.2.0). Using 3.11.14 ensures stable GPU acceleration
#   on Apple Silicon via the Metal backend.
